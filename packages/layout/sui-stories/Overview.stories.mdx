import { Meta, Source } from '@storybook/addon-docs';
import { Docs } from '@wpmudev/docs-elements';
import dedent from 'dedent';

<Meta title="SUI/CSS Framework/Layout/Overview" />

<Docs.Page title="Overview" light={true} status="ready">
	<Docs.Content
		title={
			<>
				Layout includes all necessary styling options for laying out
				your plugin, including wrapping containers, a powerful grid
				system, a flexible media object, and responsive utility classes.
			</>
		}
	>
		<Docs.Section title="Installation">
			<p>
				There are two ways to access Layout stylesheets: install the
				complete <strong>CSS Framework</strong> package or the single
				package that contains only the Layout stylesheets you need.
			</p>
			<p>
				Layout single package contains three files in different formats:
				CSS, Minified CSS, and Sassy CSS (Scss).
			</p>
			<Source
				dark
				language="shell" // bash, shell, zsh, sh
				format={false}
				code={dedent`
npm install @wpmudev/sui-css --save-dev
				`}
			/>
		</Docs.Section>
		<Docs.Section title="Usage">
			<p>
				Import layout stylesheet inside an existing SCSS file in your
				project; do the following:
			</p>
			<Source
				dark
				language="css"
				format={false}
				code={dedent`
@forward "~@wpmudev/css-layout/layout";
				`}
			/>
			<p>
				Import layout elements and modifiers individually inside an
				existing SCSS file in your project; do the following:
			</p>
			<Source
				dark
				language="css"
				format={false}
				code={dedent`
@use "~@wpmudev/css-layout/_layout/_layout-elements/layout-elements" as sui-layout--element;
@use "~@wpmudev/css-layout/_layout/_layout-modifiers/layout-modifiers" as sui-layout--modifier;\n
// Layout elements.
@include sui-layout--element.layout-main(layout);
@include sui-layout--element.layout-content(layout);\n
// Layout modifiers.
@include sui-layout--modifier.spacing-top(layout);
@include sui-layout--modifier.spacing-sides(layout);
				`}
			/>
			<Source
				dark
				language="css"
				format={false}
				code={dedent`
@use "~@wpmudev/css-layout/_grid/_grid-elements/grid-elements" as sui-grid--element;
@use "~@wpmudev/css-layout/_grid/_grid-modifiers/grid-modifiers" as sui-grid--modifier;\n
// Grid elements.
@include sui-grid--element.grid-row(row);
@include sui-grid--element.grid-col(col);\n
// Grid modifiers.
@include sui-grid--modifier.row-inline(row);
@include sui-grid--modifier.row-stacked(row);
@include sui-grid--modifier.column-size(col);
@include sui-grid--modifier.column-spacing(col);
				`}
			/>
		</Docs.Section>
	</Docs.Content>
	<Docs.Footer />
</Docs.Page>
