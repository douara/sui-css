import { Canvas, Story, Source } from '@storybook/addon-docs';
import { Docs } from '@wpmudev/docs-elements';
import dedent from 'dedent';

import HeadingTable from './components/HeadingTable.mdx';

<Docs.Page title="Heading" light={true} status="ready">
	<Docs.Content
		title={
			<>
				Headings class names are required and are the safest way to
				style your content correctly without jeopardizing the hierarchy.
			</>
		}
	>
		<Docs.Section>
			<p>
				Usually, on other CSS Frameworks, heading styles apply directly
				to the tag. Still, in the{' '}
				<strong>Shared UI CSS Framework</strong> case, we avoid using
				this method to prioritize content hierarchy for a good reading
				experience for our screen reader users.
			</p>
			<p>
				In other words, a Heading 4 tag can use, without a problem, the
				Heading 2 styles if necessary to maintain the content hierarchy
				and, simultaneously, respect the design.
			</p>
		</Docs.Section>
		<Docs.Section title="Usage">
			<p>
				To convert text into a heading, you must include the{' '}
				<code>sui-heading</code> class to your text, but this alone
				won't do much except add common heading styles such as padding,
				letter spacing, and more.
			</p>
			<Canvas>
				<div className="sui-wrap sui-theme--light">
					<div className="sui-layout sui-layout--spacing sui-layout--justify">
						<h1 className="sui-heading">The Quick Brown Fox</h1>
					</div>
				</div>
			</Canvas>
			<Source
				dark
				language="html"
				format={false}
				code={dedent`
<h1 class="sui-heading">The Quick Brown Fox</h1>
				`}
			/>
		</Docs.Section>
		<Docs.Section title="Modifiers">
			<p>
				Use a class modifier from Heading 1 (
				<code>sui-heading--h1</code>) to Heading 5 (
				<code>sui-heading--h5</code>) to style text properly. Look at
				the table below:
			</p>
			<Canvas>
				<div className="sui-wrap sui-theme--light">
					<div className="sui-layout sui-layout--spacing sui-layout--justify">
						<HeadingTable />
					</div>
				</div>
			</Canvas>
		</Docs.Section>
		<Docs.Section title="Alignment">
			<p>
				To correctly align text, you need to assign the text-align
				property to the parent container where your text stands.
			</p>
			<Canvas>
				<div className="sui-wrap sui-theme--light">
					<div
						className="sui-layout sui-layout--spacing sui-layout--justify"
						style={{ textAlign: 'center' }}
					>
						<h3 class="sui-heading sui-heading--h3">
							The Quick Brown Fox
						</h3>
					</div>
				</div>
			</Canvas>
			<p>
				In the example above, you can see a Heading 3 text that is{' '}
				<strong>center-aligned</strong>. The parent container has the{' '}
				<code>text-align: center</code> property.
			</p>
		</Docs.Section>
	</Docs.Content>
	<Docs.Footer />
</Docs.Page>
