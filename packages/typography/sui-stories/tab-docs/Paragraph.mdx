import { Canvas, Story, Source } from '@storybook/addon-docs';
import { Docs } from '@wpmudev/docs-elements';
import dedent from 'dedent';

<Docs.Page title="Paragraph" light={true} status="ready">
	<Docs.Content title="Body content or paragraph is always present in every page design.">
		<Docs.Section title="Usage">
			<p>
				Simple and basic. Add some content and wrap it with the
				paragraph tag; it will immediately adopt the necessary styles.
			</p>
			<Canvas>
				<div className="sui-wrap sui-theme--light">
					<div className="sui-layout sui-layout--spacing sui-layout--justify">
						<p>The quick brown fox jumps over a lazy dog.</p>
					</div>
				</div>
			</Canvas>
			<Source
				dark
				language="html"
				format={false}
				code={dedent`
<p>The quick brown fox jumps over a lazy dog.</p>
				`}
			/>
			<p>Above, you can find a simple example of some paragraph text.</p>
		</Docs.Section>
		<Docs.Section title="Modifiers">
			<p>
				To modify a paragraph, just make use of the official tags. For
				bold text, use <code>&lt;strong&gt;</code>; while for italic
				text, use <code>&lt;i&gt;</code> tag as shown on the example
				below.
			</p>
			<Canvas>
				<div className="sui-wrap sui-theme--light">
					<div
						className="sui-layout sui-layout--spacing sui-layout--justify"
						style={{ textAlign: 'center' }}
					>
						<p>The quick brown fox jumps over a lazy dog.</p>
						<p>
							<strong>
								The quick brown fox jumps over a lazy dog.
							</strong>
						</p>
						<p>
							<i>The quick brown fox jumps over a lazy dog.</i>
						</p>
					</div>
				</div>
			</Canvas>
		</Docs.Section>
		<Docs.Section title="Alignment">
			<p>
				If you need to align text in a different direction, assign the{' '}
				<strong>text-align</strong> property to the parent container
				where your text stands.
			</p>
			<Canvas>
				<div className="sui-wrap sui-theme--light">
					<div
						className="sui-layout sui-layout--spacing sui-layout--justify"
						style={{ textAlign: 'center' }}
					>
						<p>The quick brown fox jumps over a lazy dog.</p>
					</div>
				</div>
			</Canvas>
			<p>
				In the example above, you can see the some{' '}
				<strong>center-aligned</strong> paragraph. The parent container
				has the <code>text-align: center</code> property.
			</p>
		</Docs.Section>
	</Docs.Content>
	<Docs.Footer />
</Docs.Page>
